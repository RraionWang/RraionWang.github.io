     from flask import Flask, jsonify, request
import sqlite3

app = Flask(__name__, static_folder='../static', static_url_path='')


# 获取所有表名的API
@app.route('/api/tables', methods=['GET'])
def get_table_names():
    conn = sqlite3.connect('frame_data.db')
    cursor = conn.cursor()
    cursor.execute("SELECT name FROM sqlite_master WHERE type='table';")
    tables = cursor.fetchall()
    conn.close()
    return jsonify([table[0] for table in tables])

# 获取指定表中所有记录的帧编号API
@app.route('/api/frames', methods=['GET'])
def get_frame_numbers():
    table_name = request.args.get('table')
    if not table_name:
        return jsonify({'error': 'Table name is required.'}), 400

    conn = sqlite3.connect('frame_data.db')
    cursor = conn.cursor()
    try:
        cursor.execute(f"SELECT FrameNumber FROM {table_name}")
    except sqlite3.OperationalError as e:
        return jsonify({'error': f'Table {table_name} not found or query failed.'}), 400

    frames = cursor.fetchall()
    conn.close()

    return jsonify([frame[0] for frame in frames])

# 获取指定表和帧编号的数据API
@app.route('/api/data', methods=['GET'])
def get_frame_data():
    table_name = request.args.get('table')
    frame_number = request.args.get('frame')

    if not table_name or not frame_number:
        return jsonify({'error': 'Table name and frame number are required.'}), 400

    conn = sqlite3.connect('frame_data.db')
    cursor = conn.cursor()
    try:
        cursor.execute(f"SELECT Data FROM {table_name} WHERE FrameNumber = ?", (frame_number,))
        data = cursor.fetchone()

        if data:
            frame_data = data[0]
            formatted_data = [list(map(int, row.split())) for row in frame_data.split(';') if row]
            return jsonify(formatted_data)
        else:
            return jsonify({'error': 'No data found for the specified frame number.'}), 404
    except sqlite3.OperationalError as e:
        return jsonify({'error': f'Table {table_name} not found or query failed.'}), 400

if __name__ == '__main__':
    app.run(debug=True)

    let playInterval = null;
    let currentFrame = 0;
    let heatmapInstance = null;
    let scrollEnabled = false;
    let scrollTimeout = null;

    // 初始化 Heatmap 和 Canvas 容器
    function initializeHeatmap() {
      const heatmapContainer = document.getElementById('heatmapContainer');
      const canvasOverlay = document.getElementById('canvasOverlay');
      const heatmapWidth = parseInt(document.getElementById('heatmapWidth').value);
      const heatmapHeight = parseInt(document.getElementById('heatmapHeight').value);

      // 限制最小宽度和高度
      const minWidth = 300;
      const minHeight = 300;
      const adjustedHeatmapWidth = Math.max(heatmapWidth, minWidth);
      const adjustedHeatmapHeight = Math.max(heatmapHeight, minHeight);

      // 设置热力图容器和 canvasOverlay 的大小
      heatmapContainer.style.width = `${adjustedHeatmapWidth}px`;
      heatmapContainer.style.height = `${adjustedHeatmapHeight}px`;
      canvasOverlay.width = adjustedHeatmapWidth;
      canvasOverlay.height = adjustedHeatmapHeight;

      // 创建 heatmap 实例
      heatmapInstance = h337.create({
        container: heatmapContainer,
        maxOpacity: 0.6,
        radius: Math.min(adjustedHeatmapWidth / 10, adjustedHeatmapHeight / 10) / 2, // 动态调整 radius
        blur: 0.85
      });

      // 调整 heatmap.js 内部生成的 canvas 尺寸
      const heatmapCanvas = heatmapInstance._renderer.canvas;
      heatmapCanvas.width = adjustedHeatmapWidth;
      heatmapCanvas.height = adjustedHeatmapHeight;
    }

    // 获取所有表名并填充下拉菜单
    async function fetchTables() {
      const response = await fetch('/api/tables');
      const tables = await response.json();
      const tableSelect = document.getElementById('tableSelect');
      tables.forEach(table => {
        const option = document.createElement('option');
        option.value = table;
        option.text = table;
        tableSelect.appendChild(option);
      });
    }

    // 根据选择的表名获取所有帧编号
    async function fetchFrames(table) {
      const response = await fetch(`/api/frames?table=${table}`);
      const frames = await response.json();
      const frameSelect = document.getElementById('frameSelect');
      frameSelect.innerHTML = '<option value="">--请选择帧--</option>'; // 重置帧编号下拉菜单
      frames.forEach(frame => {
        const option = document.createElement('option');
        option.value = frame;
        option.text = frame;
        frameSelect.appendChild(option);
      });
    }

    // 根据选择的表名和帧编号获取帧数据
    async function fetchData(table, frame) {
      if (!frame) {
        console.error('Frame is not selected or invalid.');
        return;
      }

      const response = await fetch(`/api/data?table=${table}&frame=${frame}`);
      if (!response.ok) {
        console.error(`Failed to fetch data: ${response.statusText}`);
        return;
      }

      const data = await response.json();
      if (!Array.isArray(data) || !Array.isArray(data[0])) {
        console.error('Data is not in the expected format.');
        return;
      }

      renderHeatmap(data);
    }

    // 渲染 heatmap 数据
    function renderHeatmap(data) {
      const dataPoints = [];
      const numRows = data.length;
      const numCols = data[0].length;
      const heatmapWidth = parseInt(document.getElementById('heatmapWidth').value);
      const heatmapHeight = parseInt(document.getElementById('heatmapHeight').value);

      // 设置 heatmap 和 canvasOverlay 的宽高
      const adjustedHeatmapWidth = Math.max(heatmapWidth, 300);
      const adjustedHeatmapHeight = Math.max(heatmapHeight, 300);
      document.getElementById('heatmapContainer').style.width = `${adjustedHeatmapWidth}px`;
      document.getElementById('heatmapContainer').style.height = `${adjustedHeatmapHeight}px`;
      document.getElementById('canvasOverlay').width = adjustedHeatmapWidth;
      document.getElementById('canvasOverlay').height = adjustedHeatmapHeight;

      // 同步调整 heatmap.js 内部 canvas 的宽高
      const heatmapCanvas = heatmapInstance._renderer.canvas;
      heatmapCanvas.width = adjustedHeatmapWidth;
      heatmapCanvas.height = adjustedHeatmapHeight;

      // 根据数据生成 heatmap.js 所需的坐标点
      for (let row = 0; row < numRows; row++) {
        for (let col = 0; col < numCols; col++) {
          dataPoints.push({
            x: col * (adjustedHeatmapWidth / numCols) + (adjustedHeatmapWidth / numCols) / 2,  // 格子中心的x位置
            y: row * (adjustedHeatmapHeight / numRows) + (adjustedHeatmapHeight / numRows) / 2,  // 格子中心的y位置
            value: data[row][col]  // 值
          });
        }
      }

      // 设置热力图数据
      heatmapInstance.setData({
        max: 100,  // 根据你的数据调整这个值
        data: dataPoints
      });

      // 绘制数值到 Canvas
      drawTextOnCanvas(data, numRows, numCols, adjustedHeatmapWidth, adjustedHeatmapHeight);
    }

    // 绘制数值到 Canvas
    function drawTextOnCanvas(data, numRows, numCols, heatmapWidth, heatmapHeight) {
      const canvas = document.getElementById('canvasOverlay');
      const ctx = canvas.getContext('2d', { willReadFrequently: true });
      ctx.clearRect(0, 0, canvas.width, canvas.height); // 清空上一次绘制内容
      ctx.font = '16px Arial';
      ctx.fillStyle = 'black';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';

      // 绘制每个格子中的数值
      for (let row = 0; row < numRows; row++) {
        for (let col = 0; col < numCols; col++) {
          const x = col * (heatmapWidth / numCols) + (heatmapWidth / numCols) / 2;
          const y = row * (heatmapHeight / numRows) + (heatmapHeight / numRows) / 2;
          const value = data[row][col];
          ctx.fillText(value, x, y); // 在格子中心绘制文本
        }
      }
    }

    // 自动播放功能
    function playFrames() {
      const interval = parseInt(document.getElementById('intervalInput').value, 10);
      if (!interval || interval <= 0) return;

      playInterval = setInterval(() => {
        const frameSelect = document.getElementById('frameSelect');
        if (frameSelect.selectedIndex === -1) {
          console.error('No frame selected.');
          return;
        }
        currentFrame = (currentFrame + 1) % frameSelect.options.length;
        frameSelect.selectedIndex = currentFrame;
        fetchData(document.getElementById('tableSelect').value, frameSelect.value);
      }, interval);
    }

    // 停止播放功能
    function stopFrames() {
      if (playInterval) {
        clearInterval(playInterval);
        playInterval = null;
      }
    }

    // 滚轮切换功能
    function toggleScrollFrames() {
      if (scrollEnabled) {
        // 停止滚轮切换
        window.removeEventListener('wheel', handleScroll);
        scrollEnabled = false;
      } else {
        // 启用滚轮切换
        window.addEventListener('wheel', handleScroll);
        scrollEnabled = true;
      }
    }

    // 处理滚轮事件
    function handleScroll(event) {
      clearTimeout(scrollTimeout); // 重置滚轮停止计时器
      const delta = Math.sign(event.deltaY);
      const frameSelect = document.getElementById('frameSelect');

      if (delta > 0) {
        // 向下滚动，下一帧
        currentFrame = (currentFrame + 1) % frameSelect.options.length;
      } else {
        // 向上滚动，上一帧
        currentFrame = (currentFrame - 1 + frameSelect.options.length) % frameSelect.options.length;
      }

      frameSelect.selectedIndex = currentFrame;
      fetchData(document.getElementById('tableSelect').value, frameSelect.value);

      // 如果滚轮停止 500ms 后没有继续滚动，停止更新热图
      scrollTimeout = setTimeout(() => {
        console.log("停止滚动，停止更新热图。");
        stopFrames();
      }, 500); // 500 毫秒作为检测停止的延迟时间
    }

    // 当用户选择表时，获取该表中的帧编号
    document.getElementById('tableSelect').addEventListener('change', function () {
      const selectedTable = this.value;
      if (selectedTable) {
        fetchFrames(selectedTable);
      }
    });

    // 当用户选择帧时，获取该帧的数据
    document.getElementById('frameSelect').addEventListener('change', function () {
      const selectedFrame = this.value;
      const selectedTable = document.getElementById('tableSelect').value;
      if (selectedTable && selectedFrame) {
        currentFrame = this.selectedIndex;
        fetchData(selectedTable, selectedFrame);
      }
    });

    // 播放按钮点击
    document.getElementById('playButton').addEventListener('click', function () {
      playFrames();
    });

    // 停止按钮点击
    document.getElementById('stopButton').addEventListener('click', function () {
      stopFrames();
    });

    // 滚轮切换按钮点击
    document.getElementById('scrollButton').addEventListener('click', function () {
      toggleScrollFrames();
    });

    // 页面加载时获取所有表
    fetchTables();
    initializeHeatmap();
  </script>
